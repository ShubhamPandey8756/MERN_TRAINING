// ex:- var arr = ['a','b','c','d','e','f']

function reverse (s){
    var arr=s.split('');
    var len=arr.length;
    var ans=[];
    for(var r=len-1;r>=0;r--)
        {
           ans.push(arr[r]);
        }
    console.log(ans);
    
}
